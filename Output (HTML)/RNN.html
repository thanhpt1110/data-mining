In[8]:
import numpy as np
# S bc thi gian u vo
lookback = 60

# Chun b d liu cho RNN
X = []
y = []

# Xy dng cc chui u vo (input sequences) da trn `lookback` 
for i in range(lookback, len(scaled_data)):
    X.append(scaled_data[i-lookback:i, 0])  # Ly 60 bc thi gian trc 
    y.append(scaled_data[i, 0])  # Gi tr tip theo lm mc tiu

X, y = np.array(X), np.array(y)

# Reshape d liu thnh (samples, time_steps, features) cho RNN
X = np.reshape(X, (X.shape[0], X.shape[1], 1))
In[9]:
# Tch d liu thnh tp hun luyn v kim tra
train_size = int(X.shape[0] * 0.8)
X_train, X_test = X[:train_size], X[train_size:]
y_train, y_test = y[:train_size], y[train_size:]
In[10]:
from keras.models import Sequential
# Xy dng m hnh RNN
model = Sequential()

# Thm lp Input  xc nh hnh dng u vo
model.add(Input(shape=(lookback, 1)))

# Thm cc lp RNN v Dropout
model.add(SimpleRNN(units=50, activation='tanh', return_sequences=True))
model.add(Dropout(0.2))
model.add(SimpleRNN(units=50, activation='tanh', return_sequences=False))
model.add(Dropout(0.2))

# Thm lp u ra
model.add(Dense(units=1))  # D on mt gi tr (gi vng tip theo)

# Bin dch m hnh
model.compile(optimizer='adam', loss='mean_squared_error')

# Hin th cu trc m hnh
model.summary()
